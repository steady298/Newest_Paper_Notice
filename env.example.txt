# 环境变量配置示例文件
# 复制此文件的内容，设置到您的环境变量中

# Gemini API 密钥（用于翻译功能）
# 获取地址：https://aistudio.google.com/
# Windows PowerShell: $env:GEMINI_API_KEY="your-api-key-here"
# Windows CMD: set GEMINI_API_KEY=your-api-key-here
# Linux/Mac: export GEMINI_API_KEY='your-api-key-here'
GEMINI_API_KEY=your-gemini-api-key-here

# PushDeer 推送密钥（用于消息推送）
# 获取地址：https://www.pushdeer.com/ 或您的自建服务器
# Windows PowerShell: $env:PUSHDEER_KEY="your-push-key-here"
# Windows CMD: set PUSHDEER_KEY=your-push-key-here
# Linux/Mac: export PUSHDEER_KEY='your-push-key-here'
PUSHDEER_KEY=your-pushdeer-key-here
